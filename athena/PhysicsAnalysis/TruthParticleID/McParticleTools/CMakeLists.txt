# Declare the package's dependencies:

                          Control/AthenaBaseComps
                          Control/CxxUtils
                          Control/DataModel
                          Control/StoreGate
                          Event/EventKernel
                          GaudiKernel
                          Generators/GenAnalysisTools/TruthHelper

                          PhysicsAnalysis/TruthParticleID/McParticleEvent

                     src/*.cxx
